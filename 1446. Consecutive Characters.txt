class Solution {
    public int maxPower(String s) {
        char[] st = s.toCharArray();

        int i = 0;

        int count = 1;
        int max = 1;

        while(i<st.length-1){
            if(st[i] != st[i+1]){
                count = 1;
            }else {
                count++;
            }

            if(max<count){
                max = count;
            }
            i++;
        }

        return max;
    }
}
