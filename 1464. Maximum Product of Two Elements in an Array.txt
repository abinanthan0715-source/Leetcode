class Solution {
    public int maxProduct(int[] nums) {

        int first = 0;
        int value = 0;
        while(first < nums.length){
            for(int i = first+1;i<nums.length;i++){
                int cal = (nums[first] - 1)  * (nums[i] - 1);
                if(value<cal){
                    value = cal;
            }
            }
            first++;
        }
        return value;
        
    }
}
