class Solution {
    public boolean isValid(String s) {
       char[] st = s.toCharArray();
        Stack<Character> stack = new Stack<>();
for(int i = 0 ; i<st.length ; i++){
    if(st[i]=='(' || st[i]=='[' || st[i]=='{'){
        stack.push(st[i]);
    }else {
          if (stack.isEmpty()) return false;
        char current = stack.pop();
            if((st[i] != ')' && current == '('  ||
               st[i] != '}' && current == '{'  ||
               st[i] != ']' && current == '[')){
                return false;
               }
        }
}
  return stack.isEmpty();
    }
}
