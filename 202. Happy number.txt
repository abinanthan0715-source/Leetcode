class Solution {
    public boolean isHappy(int n) {

        if(n==0) return true;
        while (n != 1 && n != 4) {  // 4 is the unhappy loop
            int sum = 0;
            while(n!=0){
            int ld = n%10;
             sum = sum+(ld*ld);
            n = n/10;
            }
            n=sum;
        }
        if(n==1){
            return true;
        }else{
            return false;
        }
    }
}
