class Solution {
    public List<Integer> findDuplicates(int[] nums) {

        HashMap<Integer,Integer> hash = new HashMap<>();

        ArrayList<Integer> val = new ArrayList<>();
        for(int i = 0; i<nums.length ;i++){
            if(hash.containsKey(nums[i])){
                val.add(nums[i]);
            }
            hash.put(nums[i],i);
        }
        return val;
    }
}
