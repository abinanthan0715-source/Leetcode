class Solution {
    public String convertToBase7(int num) {
        int temp=0;  // 100
        int value = 0;
        int cnt = 0;

        while(num!=0){
            temp = (num % 7); //  2 + 0 + 2
            value = ((int)Math.pow(10,cnt) * temp) + value;
            cnt++;
            
            num = num /7;    //  14
            
        }
        String tem =  Integer.toString(value);
        return tem;
    }
}
