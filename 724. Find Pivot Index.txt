class Solution {
    public int pivotIndex(int[] nums) {
        int sum = 0;

        for(int i = 0;i<nums.length;i++){
            sum = sum+nums[i];
        }

        int right = 0;
        int left = 0;

        for(int i =0;i<nums.length;i++){
            right = sum -left- nums[i];

              if(left == right){
                return i;
              }
              left = left+nums[i];
        }
      return -1;
    }
}
