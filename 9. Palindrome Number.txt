class Solution {
    public boolean isPalindrome(int x) {
        // int x = 121;
        if(0>x){
            return false;
        }
        int rev =0;
        int temp = x;
        while(x!=0){
            int ld = x%10;
            rev = (rev*10) + ld;
            x = x/10;
        }
        x = temp;
        if(x == rev){
            return true;
        }else{
            return false;
        }
    }
}
